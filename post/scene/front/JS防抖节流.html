                        # 防抖

                        控制函数一定时间内执行的次数。意味着N秒内函数只会被执行一次，如果N秒内多次触发，则重新计算延迟时间，执行最后一次

                        应用场景：输入框、滚动、窗口大小

                        function debounce(fun, wait) {
    let timer;
    return function(...args) {
        let context=this;
        if (timer) clearTimeout(timer);

        timer = setTimeout(() => {
            fun.apply(context, args)
        }, wait)
    }
}

                        # 节流

                        规定一个单位时间内最多只能触发一次函数的执行，如果多次触发只执行第一次

                        应用场景：按钮点击，上拉触底加载更多

	resturn function(...args){
		let context=this;

				timer=null;
				fun.apply(context,args)
			},wait)
		}
	}
}
