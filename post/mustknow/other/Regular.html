                                    # 第一章 字符匹配

                                    # 模糊匹配

                                    # 横向模糊匹配

                                    一个正则可匹配的字符串长度是不固定的，实现方式是使用量词

                                    /ab{2,5}c/   //第一个字符a，接下来是2-5个字符b，最后是字符c

                                    # 纵向模糊匹配

                                    一个正则匹配的字符串，具体到某一位字符时，它可以是不确定的字符，实现方式是使用字符组

                                    /a[123]b/  //第一个字符是a，第二个字符是1、2、3中的其中一个，最后一个是b

                                    # 字符组

                                    # 简写形式

                                    字符组具体含义
                                    \d[0-9]，表示数字
                                    \D[^0-9]，表示除数字外字符
                                    \w[0-9a-zA-Z_]，表示数字、大小写字母、下划线
                                    \W[^0-9a-zA-Z_]，表示非单词字符
                                    \s[\t\v\n\r\f]，表示空白符，包括空格、水平/垂直制表符、换行父、回车符、换页符
                                    \S[^\t\v\n\r\f]，表示非空白符
                                    .表示任意字符

                                    # 范围表示法

                                    字符组字符数量过多时，可以使用范围表示法，实现方式是使用连接字符 -

                                    [1234356abcdefGHIJKLMN]   //简写，[1-6a-=fG-N]

                                    连接字符具有特殊用途，如果将其作为字符使用时，具有以下方法

                                    //假设需要匹配a、-、z三个字符

1、[-az]
2、[az-]
3、[a\-z]

                                    # 字符排除法

                                    某位字符可以是除了给定字符外的任意字符,实现方式是字符组开头使用字符 ^

                                    [^abc] //表示是a、b、c外的任意一个字符

                                    # 量词

                                    # 简写形式

                                    量词具体含义
                                    {m,}表示至少出现m次
                                    {m}表示出现m次
                                    ?等价于{0,1}，表示出现或不出现
                                    +等级于{1，}，表示至少出现一次
                                    *等级于{0，}，表示出现任意次

                                    # 贪婪匹配

                                    尽可能的匹配，只要在范围内越多越好

                                    let str = "123 1234 12345 123456"
let regex = /\d{2,5}/g

str.match(regex) //['123','1234','12345','12345']

                                    # 惰性匹配

                                    尽可能少的匹配，只要达到要求则不再往下尝试，实现方式就是在量词后添加 ?

                                    let str = "123 1234 12345 123456"
let regex = /\d{2,5}?/g

str.match(regex) //['12','12','34','12','34,'12' ,'34' , '56']

                                    # 多选分支

                                    支持多个模式选其一，实现方式是使用 |

                                    let str = "good idea,niece try"
let regx = /good|nice/g

str.match(regex) //['good','nice']

                                    分支结构是惰性的，即前面匹配则不在尝试

                                    let str = "goodby"

let regx = /good|goodbye/g
str.match(regex) //['good']

let regx = /goodbye|good/g
str.match(regex) //['goodbye']

                                    # 第二章 位置匹配

                                    符号含义
                                    ^匹配开头
                                    $匹配结尾
                                    \b单词边界
                                    \B非单词边界
                                    (?=p)p为子模式，即p前面位置
                                    (?!p)p为子模式，即非p前面位置

                                    # 位置特性

                                    对于位置的理解，我们可以理解成空字符 ""

                                    "hello" == "" + "h" + "" + "e" + "" + "l" + "" + "l" + "" + "o" + ""

//等价于
"hello" == "" + "" + "hello"

//以下正则为等价
 /^hello$/
 /^^hello$$$/
 /(?=he)^^he(?=\w)llo$\b\b$/

                                    # 开头结尾

                                    ^（脱字符）匹配开头，在多行匹配中匹配行开头

                                    $（美元符号）匹配结尾，在多行匹配中匹配行结尾

                                    var result = "hello".replace(/^|$/g, '#');
// => "#hello#"


                                    /*
修饰符
g：全局匹配
m：多行匹配
*/


var result = "I\nlove\njavascript".replace(/^|$/gm, '#');
/*
#I#
#love#
#javascript#
*/

                                    # 单词边界

                                    \b 单词边界，具体就是 \w 与 \W 之间的位置，也包括 \w 与 ^ 之间的位置，和 \w 与 $ 之间的位置

                                    var result = "[JS] Lesson_01.mp4".replace(/\b/g, '#')
// => "[#JS#] #Lesson_01#.#mp4#"

                                    \B 非单词边界，具体说来就是 \w 与 \w、 \W 与 \W、^ 与 \W，\W 与 $ 之间的位置

                                    var result = "[JS] Lesson_01.mp4".replace(/\B/g, '#')
// => "#[J#S]# L#e#s#s#o#n#_#0#1.m#p#4"

                                    # 模式边界

                                    (?=p)，其中 p 是一个子模式，即 p 前面的位置

                                    var result = "hello".replace(/(?=l)/g, '#');
// => "he#l#lo"

                                    (?1=p)，其中 p 是一个子模式，即非 p 前面的位置

                                    var result = "hello".replace(/(?!l)/g, '#');
// => "#h#ell#o#"

                                    # 验证密码

                                    密码长度 6-12 位，由数字、小写字符和大写字母组成，但必须至少包括 2 种字符

                                    一、题目转换

                                    • 同时包含数字和小写字母
                                    • 同时包含数字和大写字母
                                    • 同时包含小写字母和大写字母
                                    • 同时包含数字、小写字母和大写字母

                                    1、不考虑至少2种字符

                                    var regex = /^[0-9A-Za-z]{6,12}$/

                                    2、必须包含数字

                                    (?=.*[0-9])^，表示开头前面还有一个位置，其实也是开头，即同一个位置

                                    (?=.*[0-9])，表示任意多个字符，必须包含数字

                                    此种形式，可认为是整个字符串的一个限制条件

                                    var regex = /(?=.*[0-9])^[0-9A-Za-z]{6,12}$/

                                    3、必须包含数字和小写字母

                                    var regex = /(?=.*[0-9])(?=.*[a-z])^[0-9A-Za-z]{6,12}$/

                                    4、必须包含数字和大写字母

                                    var regex = /(?=.*[0-9])(?=.*[A-Z])^[0-9A-Za-z]{6,12}$/;

                                    5、必须包含大写字母和小写字母

                                    var regex = /(?=.*[a-z])(?=.*[A-Z])^[0-9A-Za-z]{6,12}$/;

                                    6、最终结果

                                    var regex = /((?=.*[0-9])(?=.*[a-z])|(?=.*[0-9])(?=.*[A-Z])|(?=.*[a-z])(?=.*[AZ]))^[0-9A-Za-z]{6,12}$/

                                    二、题目转换

                                    • 不能全部是数字
                                    • 不能全部是小写字母
                                    • 不能全部是大写字母

                                    1、不能全是数字

                                    var regex = /(?!^[0-9]{6,12}$)^[0-9A-Za-z]{6,12}$/

                                    2、不能全部小写字母

                                    var regex = /(?!^[a-z]{6,12}$)^[0-9A-Za-z]{6,12}$/

                                    3、不能全部大写字母

                                    var regex = /(?!^[A-Z]{6,12}$)^[0-9A-Za-z]{6,12}$/

                                    4、最终结果

                                    var regex = /(?!^[0-9]{6,12}$)(?!^[a-z]{6,12}$)(?!^[A-Z]{6,12}$)^[0-9A-Za-z]{6,12}$/

                                    # 反向引用

                                    匹配如下字符串

                                    2016-06-12
2016/06/12
2016.06.12

                                    使用的正则如下，其中 / 和 . 需要转义。虽然匹配了要求的情况，但也匹配 2016-06/12 这样的数据

                                     /\d{4}(-|\/|\.)\d{2}(-|\/|\.)\d{2}/

                                    此时要求分隔符一致，则需要使用反向引用

                                    /\d{4}(-|\/|\.)\d{2}\1\d{2}/

                                    \1 表示引用第一个分组，无论第一组匹配的结果是什么，\1 都需要匹配同样具体的某个字符

                                    \n 表示引用第n个分组

                                    特殊：\10 ，表示第十个分组，而不是 \1 和 0

                                    匹配 \1 和 0 ： （?:\1)0 或 \1(?:0)

  return c ? c.toUpperCase() : '';
  });

                                    转义

                                    // 将HTML特殊字符转换成等值的实体
function escapeHTML (str) {
  var escapeChars = {
  '<' : 'lt',
  '>' : 'gt',
  '"' : 'quot',
  '&' : 'amp',
  '\'' : '#39'
  };
  return str.replace(new RegExp('[' + Object.keys(escapeChars).join('') +']', 'g'),
function (match) {
  return '&' + escapeChars[match] + ';';
  });
}
console.log( escapeHTML('<div>Blah blah blah</div>') );
// => "&lt;div&gt;Blah blah blah&lt;/div&gt";

                                    反转义

                                    // 实体字符转换为等值的HTML。
function unescapeHTML (str) {
  var htmlEntities = {
  nbsp: ' ',
  lt: '<',
  gt: '>',
  quot: '"',
  amp: '&',
  apos: '\''
  };
  return str.replace(/\&([^;]+);/g, function (match, key) {
  if (key in htmlEntities) {
  return htmlEntities[key];
  }
  return match;
  });
}
console.log( unescapeHTML('&lt;div&gt;Blah blah blah&lt;/div&gt;') );
// => "<div>Blah blah blah</div>"

// => "2017/06.27"

                                    # match返回结果格式问题

                                    返回结果的格式与修饰符 g 相关

                                    • 无，返回标准匹配格式
                                    • 有，返回所有匹配内容

                                    没有匹配时，无关修饰符，返回null

                                    var string = "2017.06.27";
var regex1 = /\b(\d+)\b/;
var regex2 = /\b(\d+)\b/g;
console.log( string.match(regex1) );
console.log( string.match(regex2) );
// => ["2017", "2017", index: 0, input: "2017.06.27"]
// => ["2017", "06", "27"]

                                    # 全局匹配exec优于match

                                    当没有修饰符 g 时，match返回的信息多，但反之则无index信息

                                    exec可解决该问题，它能接着上次匹配后继续匹配

                                    var string = "2017.06.27";
var regex2 = /\b(\d+)\b/g;
console.log( regex2.exec(string) );
console.log( regex2.lastIndex);
console.log( regex2.exec(string) );
console.log( regex2.lastIndex);
console.log( regex2.exec(string) );
console.log( regex2.lastIndex);
console.log( regex2.exec(string) );
console.log( regex2.lastIndex);
// => ["2017", "2017", index: 0, input: "2017.06.27"]
// => 4
// => ["06", "06", index: 5, input: "2017.06.27"]
// => 7
// => ["27", "27", index: 8, input: "2017.06.27"]
// => 10
// => null
// => 0

                                    # 全局匹配起始位置问题

                                    search、match、test、exec每次匹配都会从 0 开始，即 lastIndex 属性始终不变

                                    正则实例 exec、test，全局匹配时每次匹配完成后都会修改 lastIndex

                                    var regex = /a/g;
console.log( regex.test("a"), regex.lastIndex );
console.log( regex.test("aba"), regex.lastIndex );
console.log( regex.test("ababc"), regex.lastIndex );
// => true 1
// => true 3
// => false 0

                                    var regex = /a/;
console.log( regex.test("a"), regex.lastIndex );
console.log( regex.test("aba"), regex.lastIndex );
console.log( regex.test("ababc"), regex.lastIndex );
// => true 0
// => true 0
// => true 0

                                    # test整体匹配问题

                                    test 是查看目标字符串是否存在字串，即部分匹配；整体匹配需使用添加开头结尾

                                    console.log( /123/.test("a123b") );
// => true
console.log( /^123$/.test("a123b") );
// => false
console.log( /^123$/.test("123") );
// => true

                                    # replace相关使用

                                    参数1：字符串 / 正则表达式

                                    参数2： 字符串 / 函数

                                    第二个参数是字符串时，如下字符具有特殊意义

                                    属性值描述
                                    $1,$2 ...,$99匹配第1-99个分组所捕获的文本
                                    $&匹配到的字符串文本
                                    $`匹配到的字符串的左边文本
                                    $'匹配到的字符串的右边文本
                                    $$美元符号
                                    var result = "2,3,5".replace(/(\d+),(\d+),(\d+)/, "$3=$1+$2");
console.log(result);
// => "5=2+3"

                                    var result = "2,3,5".replace(/(\d+)/g, "$&$&$&");
console.log(result);
// => "222,333,555"

                                    var result = "2+3=5".replace(/=/, "$&$`$&$'$&");
console.log(result);
// => "2+3=2+3=5=5"

                                    "1234 2345 3456".replace(/(\d)\d{2}(\d)/g, function (match, $1, $2, index, input) {
  console.log([match, $1, $2, index, input]);
});
// => ["1234", "1", "4", 0, "1234 2345 3456"]
// => ["2345", "2", "5", 5, "1234 2345 3456"]
// => ["3456", "3", "6", 10, "1234 2345 3456"]

                                    # 构造函数相关问题

                                    不推荐使用构造函数生成正则，而优先使用字面量，因为会多写 / ,一般是正则不确定需要动态生成时使用

                                    var string = "2017-06-27 2017.06.27 2017/06/27";
var regex = /\d{4}(-|\.|\/)\d{2}\1\d{2}/g;
console.log( string.match(regex) );
// => ["2017-06-27", "2017.06.27", "2017/06/27"]
regex = new RegExp("\\d{4}(-|\\.|\\/)\\d{2}\\1\\d{2}", "g");
console.log( string.match(regex) );
// => ["2017-06-27", "2017.06.27", "2017/06/27"]
