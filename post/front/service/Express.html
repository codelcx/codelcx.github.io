                  # 路由使用

                  指的是客户端请求与服务器处理函数之间的映射关系，由请求类型、请求URL、处理函数组成

                  将路由挂载到实例对象

                  app.post('/',()={})//此种方式不易管理路由

                  模块化路由

                  //创建router.js
const express = required('express')
const router = express.Router()

router.get('/',()=>{})
router.post('/',()=>{})

module.export = router

                  //使用路由模块

                  # 中间件

                  # 中间件与路由

                  区分中间件与路由，通过处理函数中是否存在next参数判断

                  app.get('/',(req,res,next)=>{next()})//必须调用next函数才能流转至下一个中间件

                  # 中间件作用

                  多个中间件之间共享同一份req、res。基于该特性可以在上有中间件中统一为req、res提娜佳自定义属性或方法，供上游中间件或路由处理

                  app.use((req,res,next)=>{
    let time = new Date()
    req.time = time
    next()
})

app.use('/',(req,res)={
    console.log(req.time)
})
//注册连续多个中间件，按照顺序流转

                  # 全局中间价

                  注册全局中间件app.use(中间件函数)

                  全局中间件：客户端发起的任何请求，到达服务器都会出触发中间件

                  app.use((req,res,next)=>{
    next()
})

                  # 注意事项

                  • 必须在路由之前注册
                  • 可连续调用多个中间件
                  • 执行中间件后，需要调用next函数
                  • 为防止代码逻辑混乱，next函数之后不再写额外代码
                  • 多个中间共享req、res

                  # 中间件分类

                  • 应用级别，绑定在app实例上
                  • 路由级别，绑定在router实例上
                  • 错误级别，捕获项目中发生的异常错误
                  • 内置中间件
                  • 第三方中间件

                  # 响应头部

                  1、Access-Controll-Allow-Origin

                  res.setHeader('Access-Controll-Allow-Origin'，'*')//通配符或具体域名

//只能设置一个具体域名，无法使用逗号分隔，如何设置多个域名白名单？
//1、设置白名单数组
//2、判断请求头req.headers.origin是否属于白名单
//3.动态添加允许请求头，es.setHeader('Access-COntroll-Allow-Origin'，req.headers.origin)

                  2、Access-Controll-Allow-Headers

                  仅支持如下9个请求头，Accept、Accept-Language、Content-Language、DPR、Downlink、Save-Data、Viewport-Width、Width、Content-Type(值限于text/plain、multipart/form-data、application/x-www-form-urlencoded)

                  //如果客户端需要向服务器发送额外请求头信息，则需要服务器配置，否则请求无效
//多个请求头之间逗号分隔
res.setHeaders('Access-Controll-AllowHeaders','Content-Type,X-Curstom-Header')

                  3、Access-Controll-Allow-Methods

                  默认情况仅支持可获得发起GET、POST、HEAD

                  //如果客户端希望支持其他请求方式，则服务器端配置
res.setHeaders('Access-Controll-Allow-Methods','POST,GET,DELETE,HEAD')

                  # 预检请求

                  浏览器与服务器正是通信之前，浏览器会先发OPTION请求进行预检，判断服务器是否允许实际请求，服务器成功响应后才会发送整整请求并携带真实数据

                  • 请求方式为GET、POST、HEAD以外的请求方法
                  • 请求头中包含自定义字段
                  • 向服务器发送aplication/json格式数据

                  # 身份认证

                  服务端渲染：Session认证(不支持跨域)

                  前后端分离：JWT认证

                  # JWT

                  三部分组成，header、payload、signature，以.分隔

                  • payload，真正用户信息，结果过加密生成的字符串
                  • header、signature安全性相关部分，保证token安全性
                  //npm install jsonwebtoken express-jwt
//jsonwebtoken:生成token字符串
//express-jwt 将jwt字符串解析还原为json对象


const express = require('express')
const jwt = require('jsonwebtoken')
const { expressjwt: expressJWT } = require('express-jwt')

const app = express()

app.use(express.json())
app.use(express.urlencoded({ extended: false }))

//为保证JWT字符串安全性，防止JWT网络传输过程中被破解，需要设置密钥
const secretKey = 'secret'

//解析token中间件，自动挂载到req.auth上
//unless，无需验证的路径
app.use(expressJWT({ secret: secretKey, algorithms: ['HS256'] }).unless({ path: [/^\/api\//] }))

//JWT错误处理中间件
app.use((err,req,res,next)=>{
    if(err.name == "UnauthorizedError"){
        return res.send({status:401,msg:'token无效'})
    }
   res.send({status:500,msg:'未知错误'})
})

//登录
app.get('/api/login', (req, res) => {
  const userInfo = req.body
  if (userInfo.username != 'admin' || userInfo.pwd != 'admin') {
    return res.send({status:1,msg:'fail'})
  }

  //加密对象，密钥，有效期,HS256(对称算法)、RS256(非对称算法)
  const token = jwt.sign({username: userInfo.username}, secretKey, { expiresIn: '30s',algorithm:'HS256'})
  console.log(token);

  //验证token
  const verify = jwt.verify(token,secretKey)
  console.log(verify);

  //解码token
  const deRes = jwt.decode(token, true)
  console.log(deRes);

  return res.send({status:0,msg:'success','Bearer'+token})
})

//测试token
app.get('/getInfo', (req, res) => {
  //添加请求头 Authorization token前添加Bearer
  return res.send(req.auth)
})
app.listen(80)

                  # session

                  cookie存储在浏览器，浏览器提供操作cookie的API，不具有安全性，因此不能将隐私数据存储在客户端

                  //npm i express-session
const express = require('express')
const session = require('express-session')

const app = express()

//请求体
app.use(express.json())
app.use(express.urlencoded({exrended:false}))

//session配置
app.use(
  session({
    secret: 'lcx',//任意字符串
    resave: false,//固定写法
    saveUninitialized:true//固定写法
  })
)

//登录
app.get('/login', (req, res) => {
  console.log(req.body);
  if (req.body.username != 'admin' || req.body.pwd != 'admin') {
    return res.send({status:1,msg:'fail'})
  }

  req.session.user = req.body//用户信息
  req.session.islogin = true//登录状态

  return res.send({status:0,msg:'success'})
})

//退出登录
app.get('/loginout', (req, res) => {
  req.session.destroy()//仅仅清空当前用户的session
  res.send({status:0,msg:'success'})
})

app.listen(80)
